                 *
                 * @param string $target Requested target attribute.
                 *
                 * @return string
                 */
                private function sanitize_link_target( $target ) {
                        $allowed = [ '_blank', '_self', '_parent', '_top' ];
                        $target  = strtolower( trim( (string) $target ) );

                        if ( ! in_array( $target, $allowed, true ) ) {
                                $target = '_self';
                        }

                        return $target;
                }

                /**
                 * Converts a raw date string into a timestamp.
                 *
                 * @param string $raw_date Date supplied through shortcode attributes.
                 *
                 * @return int|false
                 */
                private function parse_event_date( $raw_date ) {
                        $raw_date = trim( (string) $raw_date );

                        if ( '' === $raw_date ) {
                                return false;
                        }

                        $timestamp = strtotime( $raw_date );

                        return $timestamp ? (int) $timestamp : false;
                }

                /**
                 * Normalizes button styles to predictable class names.
                 *
                 * @param string $style Raw style attribute value.
                 *
                 * @return string
                 */
                private function normalize_button_style( $style ) {
                        $style = strtolower( trim( (string) $style ) );

                        if ( '' === $style ) {
                                return 'btn-primary';
                        }

                        return 'btn-' . preg_replace( '/[^a-z0-9\-]+/', '-', $style );
                }
